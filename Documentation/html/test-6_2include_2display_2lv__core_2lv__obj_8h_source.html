158{
159 struct _lv_obj_t * par; /*Pointer to the parent object*/
160 lv_ll_t child_ll; /*Linked list to store the children objects*/

164 lv_signal_func_t signal_func; /*Object type specific signal function*/

170#if LV_OBJ_FREE_PTR != 0
171 void * free_ptr; /*Application specific pointer (set it freely)*/
172#endif
173
174#if USE_LV_GROUP != 0
175 void * group_p; /*Pointer to the group of the object*/
176#endif
177 /*Attributes and states*/
178 uint8_t click :1; /*1: Can be pressed by an input device*/
179 uint8_t drag :1; /*1: Enable the dragging*/
180 uint8_t drag_throw :1; /*1: Enable throwing with drag*/
181 uint8_t drag_parent :1; /*1: Parent will be dragged instead*/
182 uint8_t hidden :1; /*1: Object is hidden*/
183 uint8_t top :1; /*1: If the object or its children is clicked it goes to the foreground*/
184 uint8_t opa_scale_en :1; /*1: opa_scale is set*/
185 uint8_t protect; /*Automatically happening actions can be prevented. 'OR'ed values from `lv_protect_t`*/
186 lv_opa_t opa_scale; /*Scale down the opacity by this factor. Effects all children as well*/
187
188 lv_coord_t ext_size; /*EXTtend the size of the object in every direction. E.g. for shadow drawing*/
189#if LV_OBJ_REALIGN
190 lv_reailgn_t realign;
191#endif

194 LV_OBJ_FREE_NUM_TYPE free_num; /*Application specific identifier (set it freely)*/
195#endif
