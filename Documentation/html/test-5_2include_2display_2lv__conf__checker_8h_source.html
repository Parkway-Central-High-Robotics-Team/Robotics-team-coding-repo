92/* VDB (Virtual Display Buffer) is an internal graphics buffer.
93 * The GUI will be drawn into this buffer first and then
94 * the buffer will be passed to your `disp_drv.disp_flush` function to
95 * copy it to your frame buffer.
96 * VDB is required for: buffered drawing, opacity, anti-aliasing and shadows
97 * Learn more: https://docs.littlevgl.com/#Drawing*/
98
99/* Size of the VDB in pixels. Typical size: ~1/10 screen. Must be >= LV_HOR_RES
100 * Setting it to 0 will disable VDB and `disp_drv.disp_fill` and `disp_drv.disp_map` functions
101 * will be called to draw to the frame buffer directly*/
102#ifndef LV_VDB_SIZE
103#define LV_VDB_SIZE ((LV_VER_RES * LV_HOR_RES) / 10)
104#endif
105

107 * Special formats are handled with `disp_drv.vdb_wr`)*/
108#ifndef LV_VDB_PX_BPP

110#endif
111
