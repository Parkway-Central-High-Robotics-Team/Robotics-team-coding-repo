385/*
386 * Registers a device in the given zero-indexed port
387 *
388 * Registers a device of the given type in the given port into the registry, if
389 * that type of device is detected to be plugged in to that port.
390 *
391 * This function uses the following values of errno when an error state is
392 * reached:
393 * ENXIO - The given value is not within the range of V5 ports (0-20), or a
394 * a different device than specified is plugged in.
395 * EADDRINUSE - The port is already registered to another device.
396 *
397 * \param port
398 * The port number to register the device
399 * \param device
400 * The type of device to register
401 *
402 * \return 1 upon success, PROS_ERR upon failure
403 */
404int registry_bind_port(uint8_t port, v5_device_e_t device_type);
405
406/*
407 * Deregisters a devices from the given zero-indexed port
408 *
409 * Removes the device registed in the given port, if there is one.
410 *
411 * This function uses the following values of errno when an error state is
412 * reached:
413 * ENXIO - The given value is not within the range of V5 ports (0-20).
414 *
415 * \param port
416 * The port number to deregister
417 *
418 * \return 1 upon success, PROS_ERR upon failure
419 */
420int registry_unbind_port(uint8_t port);
421
422/*
423 * Returns the type of device registered to the zero-indexed port.
424 *
425 * This function uses the following values of errno when an error state is
426 * reached:
427 * ENXIO - The given value is not within the range of V5 ports (0-20).
428 *
429 * \param port
430 * The V5 port number from 0-20
431 *
432 * \return The type of device that is registered into the port (NOT what is
433 * plugged in)

435v5_device_e_t registry_get_bound_type(uint8_t port);
